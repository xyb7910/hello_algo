# 选择排序

朴素版代码
```cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 1e6 + 10;
int a[N], n;
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++)
        scanf("%d", &a[i]);
    for (int i = 1; i <= n - 1; i ++)
        for (int j = i + 1; j <= n; j ++)
        {
            if(a[i] > a[j])
                swap(a[i], a[j]);
        }
    for (int i = 1; i <= n; i ++)
        printf("%d ", a[i]);
    return 0;
}
```
优化版本
```cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 1e6 + 10;
int a[N], n;
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++)
        scanf("%d", &a[i]);
    for (int i = 1; i <= n - 1; i ++) {
        int minj = i;
        for (int j = i + 1; j <= n; j ++)
        {
            if(a[minj] > a[j]) minj = j;
        }
        swap(a[i], a[minj]);
    }
    for (int i = 1; i <= n; i ++)
        printf("%d ", a[i]);
    return 0;
}
```

# 冒泡排序
朴素版代码
```cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 1e6 + 10;
int a[N], n;
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++)
        scanf("%d", &a[i]);
    for (int i = 1; i <= n - 1; i ++) {
        for (int j = 1; j <= n - i; j ++)
        {
            if(a[j] > a[j + 1]) swap(a[j], a[j + 1]); 
        }
    }
    for (int i = 1; i <= n; i ++)
        printf("%d ", a[i]);
    return 0;
}
```

优化版代码
```cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 1e6 + 10;
int a[N], n;
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++)
        scanf("%d", &a[i]);
    for (int i = 1; i <= n - 1; i ++) {
        bool f = false;
        for (int j = 1; j <= n - i; j ++)
        {
            if(a[j] > a[j + 1]) 
            {
                swap(a[j], a[j + 1]);
                f = true;
            }
        }
        if(f == false) break;
    }
    for (int i = 1; i <= n; i ++)
        printf("%d ", a[i]);
    return 0;
}
```
